namespace = SA_Count

country_event = {
	id = SA_Count.0
	is_triggered_only = yes
	hide_window = yes
	
	trigger = {
		is_ai = no
	}

	immediate = {
		set_variable = { which = count_years value = 2200 }
		set_variable = { which = count_months value = 1 }
		set_variable = { which = count_days value = 1 }
		country_event = {
			id = SA_Count.1
			days = 1
		}
	}
}

country_event = {
	id = SA_Count.1
	is_triggered_only = yes
	hide_window = yes

	immediate = {
		change_variable = { which = count_days value = 1 }
		if = {
			limit = { 
				check_variable = {which = count_days value > 30 }
			}
			change_variable = { which = count_months value = 1 }
			set_variable = { which = count_days value = 1 }
		}
		if = {
			limit = {
				check_variable = {which = count_months value > 12}
			}
			change_Variable = {which = count_years value = 1 }
			set_variable = { which = count_months value = 1 }
		}
		country_event = {
			id = SA_Count.1
			days = 1
		}
	}
}
